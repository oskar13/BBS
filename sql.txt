CREATE TABLE posts (
	post_ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	user_ID INT,
	poster_ip VARCHAR(30),
	post_date INT,
	post_pic VARCHAR(100),
	post_pic_thumb VARCHAR(100),
	post_content TEXT,
	post_stiky_level INT,
	parent_ID INT,
	board_ID INT,
	del_password VARCHAR(20),
	file_size INT,
	file_name VARCHAR(50),
	file_x INT,
	file_y INT
);

CREATE TABLE boards (
	board_ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	board_meta TEXT,
	board_url VARCHAR(15),
	board_name VARCHAR(40)
);

CREATE TABLE reply (
	r_ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	post_ID INT,
	r_parent_ID INT
);

CREATE TABLE articles (
	article_ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	user_ID INT,
	heading VARCHAR(200),
	a_date INT,
	content TEXT
);

CREATE TABLE comments (
	comment_ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	article_ID INT,
	user_ID INT,
	c_date INT,
	comment TEXT
);

CREATE TABLE users (
	user_ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	user_name VARCHAR(50),
	user_pass VARCHAR(250),
	premission_ID INT,
	banned INT,
	ban_reason TEXT
);

CREATE TABLE premissions (
	premisssion_ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	user_ID INT,
	level_ID INT,
	board_ID INT
);

CREATE TABLE level (
	level_ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	description VARCHAR(200)
);

CREATE TABLE warnings (
	warning_ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	user_ID INT,
	comment VARCHAR(200)
);
