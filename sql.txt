CREATE TABLE boards (
	board_ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	board_meta TEXT,
	board_url VARCHAR(15),
	board_name VARCHAR(40)
);

CREATE TABLE thread (
	thread_ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	user_ID INT,
	poster_ip VARCHAR(30),
	post_date INT,
	post_content TEXT,
	board_ID INT NOT NULL,
	sticky_level INT DEFAULT 0,
	del_pass VARCHAR(20),
	pic_ID INT
);

CREATE TABLE reply (
	reply_ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	user_ID INT,
	poster_ip VARCHAR(30),
	post_date INT,
	post_content TEXT,
	thread_ID INT,
	board_ID INT,
	del_pass VARCHAR(20),
	pic_ID INT
);

CREATE TABLE pictures (
	pic_ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	pic_url VARCHAR(220),
	pic_url_thumb VARCHAR(220),
	pic_size INT,
	pic_name VARCHAR(50),
	file_x INT,
	file_y INT
);

CREATE TABLE reply_quote (
	r_ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	reply_ID INT,
	r_parent_ID INT
);

CREATE TABLE thread_quote (
	t_ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	thread_ID INT,
	reply_ID INT
);





CREATE TABLE articles (
	article_ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	user_ID INT,
	heading VARCHAR(200),
	a_date INT,
	content TEXT
);

CREATE TABLE comments (
	comment_ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	article_ID INT,
	user_ID INT,
	c_date INT,
	comment TEXT
);

CREATE TABLE users (
	user_ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	user_name VARCHAR(50),
	user_pass VARCHAR(250),
	premission_ID INT DEFAULT 0,
	banned INT DEFAULT 0,
	ban_reason TEXT
);

CREATE TABLE premissions (
	premisssion_ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	user_ID INT,
	level_ID INT DEFAULT 0,
	board_ID INT NOT NULL
);

CREATE TABLE level (
	level_ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	description VARCHAR(200)
);

CREATE TABLE warnings (
	warning_ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	user_ID INT,
	comment VARCHAR(200)
);
